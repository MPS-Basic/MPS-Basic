111 void calViscosity(const double& re) {
112 double n0 = refValues.n0_forLaplacian;
113 double lambda = refValues.lambda;
114 double a = (settings.kinematicViscosity) * (2.0 * settings.dim) / (n0 * lambda);
115
116#pragma omp parallel for
117 for (auto& pi : particles) {
118 if (pi.type != ParticleType::Fluid)
119 continue;

160 void collision() {
161 for (auto& pi : particles) {
162 if (pi.type != ParticleType::Fluid)
163 continue;
164
165 for (auto& neighbor : pi.neighbors) {
166 Particle& pj = particles[neighbor.id];
167 if (pj.type == ParticleType::Fluid && pj.id >= pi.id)
168 continue;
169
170 if (neighbor.distance < settings.collisionDistance) {
171
172 double invMi = pi.inverseDensity(settings.fluidDensity);
173 double invMj = pj.inverseDensity(settings.fluidDensity);
174 double mass = 1.0 / (invMi + invMj);
175
176 Eigen::Vector3d normal = (pj.position - pi.position).normalized();
177 double relVel = (pj.velocity - pi.velocity).dot(normal);
178 double impulse = 0.0;
179 if (relVel < 0.0)
180 impulse = -(1 + settings.coefficientOfRestitution) * relVel * mass;
181 pi.velocity -= impulse * invMi * normal;
182 pj.velocity += impulse * invMj * normal;
183
184 double depth = settings.collisionDistance - neighbor.distance;
185 double positionImpulse = depth * mass;
186 pi.position -= positionImpulse * invMi * normal;
187 pj.position += positionImpulse * invMj * normal;
188

275 void setMatrix(const double& re) {
276 std::vector<Eigen::Triplet<double>> triplets;
277 auto n0 = refValues.n0_forLaplacian;
278 auto a = 2.0 * settings.dim / (n0 * refValues.lambda);
279 coefficientMatrix.resize(particles.size(), particles.size());
280
281 for (auto& pi : particles) {
282 if (pi.boundaryCondition != FluidState::Inner)
283 continue;
284
285 double coefficient_ii = 0.0;
286 for (auto& neighbor : pi.neighbors) {
287 Particle& pj = particles[neighbor.id];
288 if (pj.boundaryCondition == FluidState::Ignored)
289 continue;
290
291 if (neighbor.distance < re) {
292 double coefficient_ij = a * weight(neighbor.distance, re) / settings.fluidDensity;
293 triplets.emplace_back(pi.id, pj.id, -1.0 * coefficient_ij);
294 coefficient_ii += coefficient_ij;
295 }
296 }
297 coefficient_ii += (settings.compressibility) / (settings.dt * settings.dt);
298 triplets.emplace_back(pi.id, pi.id, coefficient_ii);
299 }
300 coefficientMatrix.setFromTriplets(triplets.begin(), triplets.end());

305 void exceptionalProcessingForBoundaryCondition() {
306 /* If there is no Dirichlet boundary condition on the fluid,
307 increase the diagonal terms of the matrix for an exception. This
308 allows us to solve the matrix without Dirichlet boundary conditions.
309 */
310 std::vector<bool> checked(particles.size(), false);
311 std::vector<bool> connected(particles.size(), false);
312
313 for (auto& pi : particles) {
314 if (pi.boundaryCondition == FluidState::FreeSurface)
315 connected[pi.id] = true;
316 if (connected[pi.id])
317 continue;
318 // BFS for connected components
319 std::queue<size_t> queue;
320 queue.push(pi.id);
321 checked[pi.id] = true;
322 while (!queue.empty()) {
323 // pop front element
324 auto v = queue.front();
325 queue.pop();
326 // search for adjacent nodes
327 for (Eigen::SparseMatrix<double, Eigen::RowMajor>::InnerIterator it(coefficientMatrix, v); it; ++it) {
328 auto nv = it.col();
329 if (!checked[nv]) {
330 if (connected[nv]) { // connected to boundary
331 connected[v] = true;
332 checked[v] = true;
333 break;
334
335 } else {
336 queue.push(nv);
337 checked[nv] = true;
338 }
339 }
340 }
341 }

349 void solveSimultaneousEquations() {
350 sourceTerm.resize(particles.size());
351 pressure.resize(particles.size());
352
353#pragma omp parallel for
354 for (auto& pi : particles) {
355 sourceTerm[pi.id] = pi.sourceTerm;
356 }
357
358 Eigen::BiCGSTAB<Eigen::SparseMatrix<double, Eigen::RowMajor>> solver;
359 solver.compute(coefficientMatrix);
360 pressure = solver.solve(sourceTerm);
361 if (solver.info() != Eigen::Success) {
362 std::cerr << "Pressure calculation failed." << std::endl;

420 void calPressureGradient(const double& re) {
421 double a = settings.dim / refValues.n0_forGradient;
422
423#pragma omp parallel for
424 for (auto& pi : particles) {
425 if (pi.type != ParticleType::Fluid)
426 continue;
427
428 Eigen::Vector3d grad = Eigen::Vector3d::Zero();
429 for (auto& neighbor : pi.neighbors) {
430 Particle& pj = particles[neighbor.id];
431 if (pj.type == ParticleType::Ghost || pj.type == ParticleType::DummyWall)
432 continue;
433
434 if (neighbor.distance < re) {
435 double w = weight(neighbor.distance, re);
436 // double dist2 = pow(neighbor.distance, 2);
437 double dist2 = (pj.position - pi.position).squaredNorm();
438 double pij = (pj.pressure - pi.minimumPressure) / dist2;
439 grad += (pj.position - pi.position) * pij * w;
440 }
441 }
442 grad *= a;
443 pi.acceleration -= grad * pi.inverseDensity(settings.fluidDensity);
444 }

488 void setNeighbors(const double& re) {
489#pragma omp parallel for
490 for (auto& pi : particles) {
491 if (pi.type == ParticleType::Ghost)
492 continue;
493
494 pi.neighbors.clear();
495
496 int ix = (int) ((pi.position.x() - domain.xMin) / bucket.length) + 1;
497 int iy = (int) ((pi.position.y() - domain.yMin) / bucket.length) + 1;
498 int iz = (int) ((pi.position.z() - domain.zMin) / bucket.length) + 1;
499
500 for (int jx = ix - 1; jx <= ix + 1; jx++) {
501 for (int jy = iy - 1; jy <= iy + 1; jy++) {
502 for (int jz = iz - 1; jz <= iz + 1; jz++) {
503 int jBucket = jx + jy * bucket.numX + jz * bucket.numX * bucket.numY;
504 int j = bucket.first[jBucket];
505
506 while (j != -1) {
507 Particle& pj = particles[j];
508
509 double dist = (pj.position - pi.position).norm();
510 if (j != pi.id && dist < re) {
511 pi.neighbors.emplace_back(j, dist);
512 }
513
514 j = bucket.next[j];
515 }
