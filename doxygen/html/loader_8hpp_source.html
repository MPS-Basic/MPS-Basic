114 std::ifstream ifs;

119 }
120
121 std::vector<Particle> particles;
122 double startTime;
123 int particleSize;
124 ifs >> startTime;
125 ifs >> particleSize;
126 for (int i = 0; i < particleSize; i++) {
127 int type_int;
128 ParticleType type;
129 Eigen::Vector3d pos, vel;
130
131 ifs >> type_int;
132 ifs >> pos.x() >> pos.y() >> pos.z();
133 ifs >> vel.x() >> vel.y() >> vel.z();
134
135 type = static_cast<ParticleType>(type_int);
136 if (type != ParticleType::Ghost) {
137 particles.emplace_back(particles.size(), type, pos, vel);
138 }
139 }
140 ifs.close();
141 ifs.clear();
142
143 return {startTime, particles};
